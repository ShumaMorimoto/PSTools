１．nodejpsのインストール
　　D:\nodejs\にインストール

２．環境設定
  $npm_root = "C:\Program Files\npm"
　npm config set prefix $npm_root
  $node_path =  npm root -g
  $browsers_path = Join-Path $npm_root "browsers"
  [System.Environment]::SetEnvironmentVariable("NODE_PATH", $node_path, "User")
  [System.Environment]::SetEnvironmentVariable("PLAYWRIGHT_BROWSERS_PATH",$browsers_path, "User")
  [System.Environment]::SetEnvironmentVariable("NODE_PATH", "$node_path", "Machine")
  [System.Environment]::SetEnvironmentVariable("PLAYWRIGHT_BROWSERS_PATH",$browsers_path, "Machine")
